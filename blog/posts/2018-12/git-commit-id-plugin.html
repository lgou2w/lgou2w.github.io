        Git-Commit-Id-Plugin

        在 Maven 和 Git 项目中使用 git-commit-id-plugin 插件打包归档让其携带有 Git commit id 的包.

        项目

        git-commit-id-plugin

        配置

        添加插件到 pom.xml 文件中.

        <!-- git-commit-id -->
<build>
<plugins>
    <plugin>
        <groupId>pl.project13.maven</groupId>
        <artifactId>git-commit-id-plugin</artifactId>
        <version>2.2.5</version>
        <executions>
            <execution>
                <id>git-info</id>
                <goals>
                    <goal>revision</goal>
                </goals>
            </execution>
        </executions>
        <configuration>
            <injectAllReactorProjects>true</injectAllReactorProjects>
            <dateFormat>yyyy-MM-dd HH:mm:ss</dateFormat> <!-- 日期格式 -->
            <verbose>true</verbose> <!-- 生成的 git.properties 包含更详情的内容 -->
            <generateGitPropertiesFile>true</generateGitPropertiesFile> <!-- 是否生成 git.properties 文件 -->
            <gitDescribe>
                <skip>false</skip>
                <always>false</always>
                <abbrev>7</abbrev> <!-- 只需要 7 位 SHA1 Commit Id -->
                <forceLongFormat>false</forceLongFormat> <!-- 是否强制长格式 -->
            </gitDescribe>
        </configuration>
    </plugin>
</plugins>
</build>
